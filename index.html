A simple way to display beautiful popupS.

Alert


×
Hello World! 
   popupS.window({
        mode: 'alert',
        content: 'Hello World!'
    });

Confirm


×
Continue? 
   popupS.window({
        mode: 'confirm',
        content: 'Continue?',
        onSubmit: function() {
                popupS.window({
                        mode: 'alert',
                        content: 'Great!'
                });
        }
    });

Prompt


×
What is your name? 
   popupS.window({
        mode: 'prompt',
        content: 'What is your name?',
        onSubmit: function(val) {
                popupS.window({
                        mode: 'alert',
                        content: 'Hi, ' + val + '!'
                });
        }
    });
