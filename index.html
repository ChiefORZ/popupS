×
Hello World! 
   popupS.alert({

Confirm


×
Continue? 
   popupS.confirm({
        content: 'Continue?',
        onSubmit: function() {
                popupS.alert({
                        content: 'Great!'
                });
        }
    });

Prompt


×
What is your name? 
   popupS.prompt({
        content: 'What is your name?',
        onSubmit: function(val) {
                popupS.alert({
                        content: 'Hi, ' + val + '!'
                });
        }
    });
